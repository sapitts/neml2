258 template <typename T = Model, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
259 T & register_model(const std::string & name, bool merge_input = true)
260 {
261 auto model_name =
262 input_options().contains(name) ? input_options().get<std::string>(name) : name;
263 if (model_name == this->name())
264 throw SetupException("Model named '" + this->name() +
265 "' is trying to register itself as a sub-model. This is not allowed.");
266
267 OptionSet extra_opts;
268 extra_opts.add_private<NEML2Object *>("_host", host());
269
270 if (!host()->factory())
271 throw SetupException("Internal error: Host object '" + host()->name() +
272 "' does not have a factory set.");
273 auto model = host()->factory()->get_object<T>("Models", model_name, extra_opts);
274
275 register_model(model, merge_input);
276 return *model;
277 }

290 template <typename T = Model, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
291 void register_model(const std::shared_ptr<T> & model, bool merge_input = true)
292 {
293 if (std::find(_registered_models.begin(), _registered_models.end(), model) !=
294 _registered_models.end())
295 throw SetupException("Model named '" + model->name() + "' has already been registered.");
296
297 if (merge_input)
298 for (auto && [name, var] : model->input_variables())
299 if (input_variables().find(name) == input_variables().end())
300 clone_input_variable(*var);
301
302 _registered_models.push_back(model);
303 }
