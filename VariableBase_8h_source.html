295// Everything below is just for convenience: We just forward operations to the the variable values
296// so that we can do
297//
298// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
299// var4 = (var1 - var2) * var3
300// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
301//
302// instead of the (ugly?) expression below
303//
304// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
305// var4 = (var1.v - var2.v) * var3.v
306// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
307#define FWD_VARIABLE_BINARY_OP(op) \
308 template <typename T1, \
309 typename T2, \
310 typename = typename std::enable_if_t<std::is_base_of_v<VariableBase, T1> || \
311 std::is_base_of_v<VariableBase, T2>>> \
312 auto op(const T1 & a, const T2 & b) \
313 { \
314 if constexpr (std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
315 return op(a(), b()); \
316 \
317 if constexpr (std::is_base_of_v<VariableBase, T1> && !std::is_base_of_v<VariableBase, T2>) \
318 return op(a(), b); \
319 \
320 if constexpr (!std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
321 return op(a, b()); \
322 } \
323 static_assert(true)
324
325FWD_VARIABLE_BINARY_OP(operator+);
326FWD_VARIABLE_BINARY_OP(operator-);
327FWD_VARIABLE_BINARY_OP(operator*);
328FWD_VARIABLE_BINARY_OP(operator/);

330FWD_VARIABLE_BINARY_OP(operator>);
331FWD_VARIABLE_BINARY_OP(operator<);
332FWD_VARIABLE_BINARY_OP(operator>=);
333FWD_VARIABLE_BINARY_OP(operator<=);
334FWD_VARIABLE_BINARY_OP(operator&&);
